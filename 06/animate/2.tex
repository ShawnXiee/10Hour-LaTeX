\newcommand{\blinker}[1]{
\begin{tikzpicture}[xscale=0.27,yscale=-0.27]
\foreach \y/\x in {#1}{
     \filldraw[red!60,draw=gray] (\x-0.9,\y-0.9) rectangle(\x-0.1,\y-0.1);
}
\draw[gray] (0,0) grid (9,9);
\end{tikzpicture}
}

\begin{animateinline}[loop, autoplay, poster = first]{5}

\end{animateinline}
